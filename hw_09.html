r_squared = NULL

for(i in 1:10){
  
  q = data.frame(o[(i + 1):4277] , o[1:(4277 - i)])

  colnames(q) = c("Today Price" , "Price in i days before")

  fit = lm(q$`Today Price` ~ q$`Price in i days before`)
  
  r_squared = c(r_squared , summary(fit)$r.squared)
